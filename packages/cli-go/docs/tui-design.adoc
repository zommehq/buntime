== Overview

Design document for the Buntime CLI TUI using the Charm stack:

* **Bubble Tea** - Elm-inspired TUI framework
* **Bubbles** - Pre-built components (text input, lists, spinners, tables, etc.)
* **Lip Gloss** - CSS-like styling for terminal apps
* **Harmonica** - Spring-based animations

=== Terminal Size Handling

[source,go]
----
func (m Model) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
    switch msg := msg.(type) {
    case tea.WindowSizeMsg:
        m.width = msg.Width
        m.height = msg.Height
        m.recalculateLayout()
    }
    return m, nil
}

func (m *Model) recalculateLayout() {
    // Adjust table columns based on width
    if m.width < 60 {
        m.appTable.SetColumns(compactColumns)
    } else if m.width > 100 {
        m.appTable.SetColumns(wideColumns)
    } else {
        m.appTable.SetColumns(normalColumns)
    }
}
